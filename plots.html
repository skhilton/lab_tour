      df = []

      ​
      # turn preference lists into `scipy.arrays`
      for site, p in prefs.items():
          prefs[site] = scipy.array([p[aa] for aa in amino_acids])
          prefs_beta[site] = scipy.power(prefs[site], beta) / scipy.power(prefs[site], beta).sum()
          
      # calculate the distance between all pairs of sites
      for pair in itertools.combinations(prefs.keys(), 2):
          assert pair[0] < pair[1]
          p1 = prefs[pair[0]]
          p2 = prefs[pair[1]]
          p1_beta = prefs_beta[pair[0]]
          p2_beta = prefs_beta[pair[1]]
          for distance in distances: 
              d = modeladequacy.prefDistance(p1, p2, distance, check_input=False)
              df.append((pair[0], pair[1], distance, d, "raw"))
              d = modeladequacy.prefDistance(p1_beta, p2_beta, distance, check_input=False)
              df.append((pair[0], pair[1], distance, d, "scaled"))

      matches = []
      for name,group in df.groupby("metric"):
          max_d = group.quantile(q=1.0, axis=0,)["distance"]
          min_d = group.quantile(q=0.5, axis=0,)["distance"]
          group = group.copy()
          group["site_pair"] = group[["site1", "site2"]].apply(lambda x: "{0}_{1}".format(x[0], x[1]), axis=1)
          group["target"] = group["distance"].apply(lambda x: "yes" if min_d <= x <= max_d else "no")
          matches.append(group)
      matches = pd.concat(matches)
      matches = matches[matches.target=="yes"].pivot(index='site_pair', columns='metric', values='target')
      matches = matches.reset_index()
      matches["number_metrics"] = matches[distances].apply(lambda x: list(x).count("yes"), axis=1)
